254  {

304  void clear(bool force=false)
305  {
306  if (force)
307  {
308  while (current_line < n_lines-1)
309  {
310  stream_out << "\033[B";
311  current_line++;
312  };
313 
314  if (current_line>0)
315  {
316  for (unsigned int i=0; i<n_lines; ++i)
317  {
318  stream_out << "\r" << std::setfill(' ') << std::setw(width) << " " << "\r" << "\033[A";
319  current_line--;
320  if (current_line==0)
321  break;
322  }
323  stream_out << "\r" << std::setfill(' ') << std::setw(width) << " " << "\r";
324  }
325  else
326  {
327  stream_out << "\r" << std::setfill(' ') << std::setw(width) << " " << "\r";
328  }
329  clear_next = false;
330  }
331  else if (clear_next)
332  {
333  stream_out << "\r" << std::setfill(' ') << std::setw(width) << " " << "\r";
334  stream_out << "\033[A" << std::endl;
335  clear_next = false;
336  }
