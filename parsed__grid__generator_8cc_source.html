308  default_create_grid( Triangulation<dim,spacedim> &tria,
309  std::string grid_name,
310  Point<spacedim> point_option_one,
311  Point<spacedim> point_option_two,

327  colorize);

329  else if (grid_name == "file")
330  {
331  GridIn<dim, spacedim> gi;

333 
334  std::ifstream in(input_grid_file_name.c_str());

336 
337  std::string ext = extension(input_grid_file_name);
338  if (ext == "vtk")
339  gi.read_vtk(in);
340  else if (ext == "msh")
341  gi.read_msh(in);
342  else if (ext == "ucd" || ext == "inp")
343  gi.read_ucd(in);
344  else if (ext == "unv")
345  gi.read_unv(in);
346  else
347  Assert(false, ExcNotImplemented());
348  }
349  // silence warnings of unused variables
350  (void)double_option_one;
351  (void)double_option_two;
352  (void)double_option_three;
353  (void)un_int_option_one;

675 template <int dim, int spacedim>
676 typename Triangulation<dim,spacedim>::MeshSmoothing
677 ParsedGridGenerator<dim, spacedim>::get_smoothing()
678 {
679  int smoothing = 0;
680  std::vector<std::string> ss = Utilities::split_string_list(mesh_smoothing);
681  for (std::string s : ss)
682  {
683  if (s == "none")
684  smoothing = smoothing|(int)Triangulation<dim,spacedim>::none;
685  else if (s == "limit_level_difference_at_vertices")
686  smoothing = smoothing|(int)Triangulation<dim,spacedim>::limit_level_difference_at_vertices;
687  else if (s == "eliminate_unrefined_islands")
688  smoothing = smoothing|(int)Triangulation<dim,spacedim>::eliminate_unrefined_islands;
689  else if (s == "patch_level_1")
690  smoothing = smoothing|(int)Triangulation<dim,spacedim>::patch_level_1;
691  else if (s == "coarsest_level_1")
692  smoothing = smoothing|(int)Triangulation<dim,spacedim>::coarsest_level_1;
693  else if (s == "allow_anisotropic_smoothing")
694  smoothing = smoothing|(int)Triangulation<dim,spacedim>::allow_anisotropic_smoothing;
695  else if (s == "eliminate_refined_inner_islands")
696  smoothing = smoothing|(int)Triangulation<dim,spacedim>::eliminate_refined_inner_islands;
697  else if (s == "eliminate_refined_boundary_islands")
698  smoothing = smoothing|(int)Triangulation<dim,spacedim>::eliminate_refined_boundary_islands;
699  else if (s == "do_not_produce_unrefined_islands")
700  smoothing = smoothing|(int)Triangulation<dim,spacedim>::do_not_produce_unrefined_islands;
701  else if (s == "smoothing_on_refinement")
702  smoothing = smoothing|(int)Triangulation<dim,spacedim>::smoothing_on_refinement;
703  else if (s == "smoothing_on_coarsening")
704  smoothing = smoothing|(int)Triangulation<dim,spacedim>::smoothing_on_coarsening;
705  else if (s == "maximum_smoothing")
706  smoothing = smoothing|(int)Triangulation<dim,spacedim>::maximum_smoothing;
707  else
708  Assert(false, ExcInternalError("Found a non supported smoothing. Should not happen:"+s))
709  }
710  return static_cast<typename Triangulation<dim,spacedim>::MeshSmoothing>(smoothing);
711 }
