50 void
51 ParameterAcceptor::initialize(const std::string filename,
52  const std::string out_filename)
53 {
54  prm.clear();
55  declare_all_parameters(prm);
56  if (filename != "")
57  {
58  // check the extension of input file
59  if (filename.substr(filename.find_last_of(".") + 1) == "prm")
60  prm.read_input(filename);
61  else if (filename.substr(filename.find_last_of(".") + 1) == "xml")
62  {
63  std::ifstream is(filename);
64  prm.read_input_from_xml(is);
65  }
66  else
67  AssertThrow(false, ExcMessage("Invalid extension of parameter file. Please use .prm or .xml"));
68  }
69  parse_all_parameters(prm);
70  if (out_filename != "")
71  {
72  std::ofstream outfile(out_filename.c_str());
73  Assert(outfile, ExcIO());
74  std::string extension = out_filename.substr(out_filename.find_last_of(".") + 1);
75 
76  if ( extension == "prm")
77  {
78  outfile << "# Parameter file generated with " << std::endl
79  << "# D2K_GIT_BRANCH= " << D2K_GIT_BRANCH << std::endl
80  << "# D2K_GIT_SHORTREV= " << D2K_GIT_SHORTREV << std::endl
81  << "# DEAL_II_GIT_BRANCH= " << DEAL_II_GIT_BRANCH << std::endl
82  << "# DEAL_II_GIT_SHORTREV= " << DEAL_II_GIT_SHORTREV << std::endl;
83  prm.print_parameters(outfile, ParameterHandler::ShortText);
84  }
85  else if (extension == "xml")
86  prm.print_parameters(outfile, ParameterHandler::XML);
87  else if (extension == "latex" || extension == "tex")
88  prm.print_parameters(outfile, ParameterHandler::LaTeX);
89  else
90  AssertThrow(false,ExcNotImplemented());
91  }
92 }

139 void ParameterAcceptor::parse_parameters(ParameterHandler &prm)
140 {
141  for (auto it = parameters.begin(); it != parameters.end(); ++it)
142  {
143  if (it->second.type() == typeid(std::string *))
144  {
145  *(boost::any_cast<std::string *>(it->second)) = prm.get(it->first);
146  }
147  else if (it->second.type() == typeid(double *))
148  {
149  *(boost::any_cast<double *>(it->second)) = prm.get_double(it->first);
150  }
151  else if (it->second.type() == typeid(int *))
152  {
153  *(boost::any_cast<int *>(it->second)) = prm.get_integer(it->first);
154  }
155  else if (it->second.type() == typeid(unsigned int *))
156  {
157  *(boost::any_cast<unsigned int *>(it->second)) = prm.get_integer(it->first);
158  }
159  else if (it->second.type() == typeid(bool *))
160  {
161  *(boost::any_cast<bool *>(it->second)) = prm.get_bool(it->first);
162  }
163  // Here we have the difficult types...
164  // First all point types.
165  else if (it->second.type() == typeid(Point<1> *))
166  {
167  std::vector<double> p =
168  Utilities::string_to_double(Utilities::split_string_list(prm.get(it->first)));
169  AssertDimension(p.size(), 1);
170 
171  Point<1> &pp = *(boost::any_cast<Point<1>*>(it->second));
172  pp[0] = p[0];
173  }
174  else if (it->second.type() == typeid(Point<2> *))
175  {
176  std::vector<double> p =
177  Utilities::string_to_double(Utilities::split_string_list(prm.get(it->first)));
178  AssertDimension(p.size(), 2);
179 
180  Point<2> &pp = *(boost::any_cast<Point<2>*>(it->second));
181  pp[0] = p[0];
182  pp[1] = p[1];
183  }
184  else if (it->second.type() == typeid(Point<3> *))
185  {
186  std::vector<double> p =
187  Utilities::string_to_double(Utilities::split_string_list(prm.get(it->first)));
188  AssertDimension(p.size(), 3);
189 
190  Point<3> &pp = *(boost::any_cast<Point<3>*>(it->second));
191  pp[0] = p[0];
192  pp[1] = p[1];
193  pp[2] = p[2];
194  }
195  else if (it->second.type() == typeid(std::vector<std::string> *))
196  {
197  std::vector<std::string> &string_list = *(boost::any_cast<std::vector<std::string>*>(it->second));
198  string_list = Utilities::split_string_list(prm.get(it->first));
199  }
200  else if (it->second.type() == typeid(std::vector<unsigned int> *))
201  {
202  std::vector<unsigned int> &int_list = *(boost::any_cast<std::vector<unsigned int>*>(it->second));
203  std::vector<std::string> string_list = Utilities::split_string_list(prm.get(it->first));
204  int_list.resize(string_list.size());
205  for (unsigned int i=0; i<string_list.size(); ++i)
206  {
207  std::istringstream reader(string_list[i]);
208  reader >> int_list[i];// = std::stoul(string_list[i]);
209  }
210  }
211  else if (it->second.type() == typeid(std::vector<double> *))
212  {
213  std::vector<double> &double_list = *(boost::any_cast<std::vector<double>*>(it->second));
214  double_list = Utilities::string_to_double(Utilities::split_string_list(prm.get(it->first)));
215  }
216  else if (it->second.type() == typeid(std::vector<std::vector<unsigned int> > *))
217  {
218  std::vector<std::vector<unsigned int> > &int_table =
219  *(boost::any_cast<std::vector<std::vector<unsigned int> >*>(it->second));
220 
221  std::vector<std::string> string_list_of_lists = Utilities::split_string_list(prm.get(it->first), ';');
222  int_table.resize(string_list_of_lists.size());
223  for (unsigned int i=0; i<int_table.size(); ++i)
224  {
225  std::vector<std::string> string_list = Utilities::split_string_list(string_list_of_lists[i], ',');
226  int_table[i].resize(string_list.size());
227  for (unsigned int j=0; j<int_table[i].size(); ++j)
228  {
229  std::istringstream reader(string_list[j]);
230  reader >> int_table[i][j];
231  }
232  }
233  }
234  else
235  {
236  AssertThrow(false, ExcNotImplemented());
237  }
238  }
239 }
