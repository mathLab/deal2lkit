53 void
54 ParameterAcceptor::initialize(const std::string filename,
55  const std::string out_filename)
56 {
57  prm.clear();
58  declare_all_parameters(prm);
59  if (filename != "")
60  {
61  // check the extension of input file
62  if (filename.substr(filename.find_last_of(".") + 1) == "prm")
63  prm.read_input(filename);
64  else if (filename.substr(filename.find_last_of(".") + 1) == "xml")
65  {
66  std::ifstream is(filename);
67  prm.read_input_from_xml(is);
68  }
69  else
70  AssertThrow(false, ExcMessage("Invalid extension of parameter file. Please use .prm or .xml"));
71  }
72  parse_all_parameters(prm);
73  if (out_filename != "")
74  {
75  std::ofstream outfile(out_filename.c_str());
76  Assert(outfile, ExcIO());
77  std::string extension = out_filename.substr(out_filename.find_last_of(".") + 1);
78 
79  if ( extension == "prm")
80  {
81  outfile << "# Parameter file generated with " << std::endl
82  << "# D2K_GIT_BRANCH= " << D2K_GIT_BRANCH << std::endl
83  << "# D2K_GIT_SHORTREV= " << D2K_GIT_SHORTREV << std::endl
84  << "# DEAL_II_GIT_BRANCH= " << DEAL_II_GIT_BRANCH << std::endl
85  << "# DEAL_II_GIT_SHORTREV= " << DEAL_II_GIT_SHORTREV << std::endl;
86  prm.print_parameters(outfile, ParameterHandler::ShortText);
87  }
88  else if (extension == "xml")
89  prm.print_parameters(outfile, ParameterHandler::XML);
90  else if (extension == "latex" || extension == "tex")
91  prm.print_parameters(outfile, ParameterHandler::LaTeX);
92  else
93  AssertThrow(false,ExcNotImplemented());
94  }
95 }

118 void ParameterAcceptor::parse_all_parameters(ParameterHandler &prm)
119 {
120  std::vector<std::string> old_secs;
121  for (unsigned int i=0; i< class_list.size(); ++i)
122  if (class_list[i] != NULL)
123  {
124  std::vector<std::string> secs;
125  secs = Utilities::split_string_list(class_list[i]->get_section_name(), sep);
126  const unsigned int secs_size = secs.size();
127 
128  if (secs_size == 1) // add a subsection with relative path
129  {
130  // enter the old path
131  for (unsigned int s=0; s<old_secs.size(); ++s)
132  prm.enter_subsection(old_secs[s]);
133 
134  // enter current subsection
135  prm.enter_subsection(secs[0]);
136 
137  class_list[i]->parse_parameters(prm);
138  class_list[i]->parse_parameters_call_back();
139 
140  // leave current subsection
141  prm.leave_subsection();
142 
143  // leave the old path
144  for (unsigned int s=0; s<old_secs.size(); ++s)
145  prm.leave_subsection();
146 
147  // the old path is unchanged
148  }
149 
150  else // we have at least one separator "/"
151  {
152  if (secs[0]=="") // the section name begins with "/"
153  {
154  // first of all remove the "empty" section
155  secs.erase(secs.begin());
156 
157  // this means absolute path so we do not enter
158  // the old path
159  for (unsigned int s=0; s<secs.size(); ++s)
160  prm.enter_subsection(secs[s]);
161 
162  class_list[i]->parse_parameters(prm);
163  class_list[i]->parse_parameters_call_back();
164 
165  // leave all the current sections
166  for (unsigned int s=0; s<secs.size(); ++s)
167  prm.leave_subsection();
168 
169  // since we have started from the root section i.e. "/"
170  // we need to set the old path to the current one
171  // without the last subsection
172 
173  // remove the last subsection
174  secs.pop_back();
175 
176  // update the old path
177  old_secs.swap(secs);
178 
179  }
180  else // append the path to the old path
181  {
182  // enter the old path
183  for (unsigned int s=0; s<old_secs.size(); ++s)
184  prm.enter_subsection(old_secs[s]);
185 
186  // new path
187  for (unsigned int s=0; s<secs.size(); ++s)
188  prm.enter_subsection(secs[s]);
189 
190  class_list[i]->parse_parameters(prm);
191  class_list[i]->parse_parameters_call_back();
192 
193  // leave all subsections
194  for (unsigned int s=0; s<(secs.size()+old_secs.size()); ++s)
195  prm.leave_subsection();
196 
197  // we need now to append the secs path (without the last subsection)
198  // to the old path
199 
200  // remove last subsection
201  secs.pop_back();
202 
203  // append secs to old_secs
204  old_secs.insert(old_secs.begin(), secs.begin(), secs.end());
205 
206  }
207  }
208  }

302 void ParameterAcceptor::parse_parameters(ParameterHandler &prm)
303 {
304  for (auto it = parameters.begin(); it != parameters.end(); ++it)
305  {
306  if (it->second.type() == typeid(std::string *))
307  {
308  *(boost::any_cast<std::string *>(it->second)) = prm.get(it->first);
309  }
310  else if (it->second.type() == typeid(double *))
311  {
312  *(boost::any_cast<double *>(it->second)) = prm.get_double(it->first);
313  }
314  else if (it->second.type() == typeid(int *))
315  {
316  *(boost::any_cast<int *>(it->second)) = prm.get_integer(it->first);
317  }
318  else if (it->second.type() == typeid(unsigned int *))
319  {
320  *(boost::any_cast<unsigned int *>(it->second)) = prm.get_integer(it->first);
321  }
322  else if (it->second.type() == typeid(bool *))
323  {
324  *(boost::any_cast<bool *>(it->second)) = prm.get_bool(it->first);
325  }
326  // Here we have the difficult types...
327  // First all point types.
328  else if (it->second.type() == typeid(Point<1> *))
329  {
330  std::vector<double> p =
331  Utilities::string_to_double(Utilities::split_string_list(prm.get(it->first)));
332  AssertDimension(p.size(), 1);
333 
334  Point<1> &pp = *(boost::any_cast<Point<1>*>(it->second));
335  pp[0] = p[0];
336  }
337  else if (it->second.type() == typeid(Point<2> *))
338  {
339  std::vector<double> p =
340  Utilities::string_to_double(Utilities::split_string_list(prm.get(it->first)));
341  AssertDimension(p.size(), 2);
342 
343  Point<2> &pp = *(boost::any_cast<Point<2>*>(it->second));
344  pp[0] = p[0];
345  pp[1] = p[1];
346  }
347  else if (it->second.type() == typeid(Point<3> *))
348  {
349  std::vector<double> p =
350  Utilities::string_to_double(Utilities::split_string_list(prm.get(it->first)));
351  AssertDimension(p.size(), 3);
352 
353  Point<3> &pp = *(boost::any_cast<Point<3>*>(it->second));
354  pp[0] = p[0];
355  pp[1] = p[1];
356  pp[2] = p[2];
357  }
358  else if (it->second.type() == typeid(std::vector<std::string> *))
359  {
360  std::vector<std::string> &string_list = *(boost::any_cast<std::vector<std::string>*>(it->second));
361  string_list = Utilities::split_string_list(prm.get(it->first));
362  }
363  else if (it->second.type() == typeid(std::vector<unsigned int> *))
364  {
365  std::vector<unsigned int> &int_list = *(boost::any_cast<std::vector<unsigned int>*>(it->second));
366  std::vector<std::string> string_list = Utilities::split_string_list(prm.get(it->first));
367  int_list.resize(string_list.size());
368  for (unsigned int i=0; i<string_list.size(); ++i)
369  {
370  std::istringstream reader(string_list[i]);
371  reader >> int_list[i];// = std::stoul(string_list[i]);
372  }
373  }
374  else if (it->second.type() == typeid(std::vector<double> *))
375  {
376  std::vector<double> &double_list = *(boost::any_cast<std::vector<double>*>(it->second));
377  double_list = Utilities::string_to_double(Utilities::split_string_list(prm.get(it->first)));
378  }
379  else if (it->second.type() == typeid(std::vector<std::vector<unsigned int> > *))
380  {
381  std::vector<std::vector<unsigned int> > &int_table =
382  *(boost::any_cast<std::vector<std::vector<unsigned int> >*>(it->second));
383 
384  std::vector<std::string> string_list_of_lists = Utilities::split_string_list(prm.get(it->first), ';');
385  int_table.resize(string_list_of_lists.size());
386  for (unsigned int i=0; i<int_table.size(); ++i)
387  {
388  std::vector<std::string> string_list = Utilities::split_string_list(string_list_of_lists[i], ',');
389  int_table[i].resize(string_list.size());
390  for (unsigned int j=0; j<int_table[i].size(); ++j)
391  {
392  std::istringstream reader(string_list[j]);
393  reader >> int_table[i][j];
394  }
395  }
396  }
397  else
398  {
399  AssertThrow(false, ExcNotImplemented());
400  }
401  }
