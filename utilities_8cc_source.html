128 bool copy_files(const std::string &files, const std::string &destination)
129 {
130  create_directory("./"+destination);
131  bool result = true;
132  int status;
133  std::vector<std::string> strs;
134  std::string new_file;
135  strs = dealii::Utilities::split_string_list(files, ' ');
136  for (size_t i = 0; i < strs.size(); i++)
137  {
138  Assert(file_exists(strs[i]), ExcMessage("Invalid name of file"));
139  new_file = destination+"/"+strs[i];
140  std::string cmd = "cp " + strs[i] + " " + new_file;
141  status = std::system( cmd.c_str() );
142  AssertThrow(status == 0, ExcCannottExecuteCommand(cmd));
143  result &= file_exists(new_file);
144  }
145  return result;
146 }
