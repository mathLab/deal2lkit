85 template <int dim, int spacedim>
86 void ParsedDataOut<dim,spacedim>::parse_parameters_call_back()
87 {
88  if ( incremental_run_prefix != "" )
89  {
90  path_solution_dir = get_next_available_directory_name(incremental_run_prefix);
91  // The use of the barrier is
92  // to avoid the case of a processor below the master node.
93 #ifdef DEAL_II_WITH_MPI
94  MPI_Barrier(comm);
95 #endif
96  if ( Utilities::MPI::this_mpi_process(MPI_COMM_WORLD) == 0)
97  create_directory(path_solution_dir);
98 #ifdef DEAL_II_WITH_MPI
99  MPI_Barrier(comm);
100 #endif
101  path_solution_dir += "/";
102  }
103  else
104  {
105  path_solution_dir = "./";
106  }
107 }

109 template <int dim, int spacedim>
110 void ParsedDataOut<dim,spacedim>::prepare_data_output(const DoFHandler<dim,spacedim> &dh,
111  const std::string &suffix)
112 {
113  deallog.push("PrepareOutput");
114  data_out = SP(new DataOut<dim, DoFHandler<dim, spacedim> > ());
115  data_out->set_default_format(DataOutBase::parse_output_format(output_format));
116 
117  current_name = path_solution_dir + base_name + suffix;
118 
119  std::string fname = current_name;
120 
121  if (data_out->default_suffix() != "")
122  {
123 
124  // If the output is needed and we have many processes, just output
125  // the one we need *in intermediate format*.
126  if (n_mpi_processes > 1)
127  {
128  fname += ("." + Utilities::int_to_string(this_mpi_process, 2) +
129  "." + Utilities::int_to_string(n_mpi_processes, 2) +
130  data_out->default_suffix()) ;
131  }
132  else
133  {
134  fname += data_out->default_suffix();
135  }
136 
137  deallog << "Will write on file: " << fname.c_str() << std::endl;
138  output_file.open(fname.c_str());
139  AssertThrow(output_file, ExcIO());
140  data_out->attach_dof_handler (dh);
141 
142  if (n_mpi_processes > 1)
143  {
144  // Output the partitioning
145  if (output_partitioning)
146  {
147  deallog << "Writing partitioning" << std::endl;
148  Vector<float> partitioning(dh.get_tria().n_active_cells());
149  for (unsigned int i=0; i<partitioning.size(); ++i)
150  partitioning(i) = this_mpi_process;
151  static Vector<float> static_partitioning;
152  static_partitioning.swap(partitioning);
153  data_out->add_data_vector (static_partitioning, "partitioning");
154  }
155  }
156  }
157  initialized = true;
158  deallog.pop();
159 }

163 template <int dim, int spacedim>
164 void ParsedDataOut<dim,spacedim>::write_data_and_clear( const Mapping<dim,spacedim> &mapping)
165 {
166  if (files_to_save!="")
167  {
168  std::vector<std::string> files;
169  files = Utilities::split_string_list(files_to_save, '%');
170  for (unsigned int i = 0; i<files.size(); ++i)
171  {
172  if (this_mpi_process == 0)
173  copy_files(files[i], path_solution_dir);
174  }
175  }
176  AssertThrow(initialized, ExcNotInitialized());
177  AssertThrow(output_file, ExcIO());
178  deallog.push("WritingData");
179  if (data_out->default_suffix() != "")
180  {
181  data_out->build_patches(mapping, subdivisions,
182  DataOut<dim, DoFHandler<dim,spacedim> >::curved_inner_cells);
183  data_out->write(output_file);
184  deallog << "Wrote output file." << std::endl;
185 
186  if (this_mpi_process == 0 && n_mpi_processes > 1 && data_out->default_suffix() == ".vtu")
187  {
188  std::vector<std::string> filenames;
189  for (unsigned int i=0; i<n_mpi_processes; ++i)
190  filenames.push_back (current_name +
191  "." + Utilities::int_to_string(i, 2) +
192  "." + Utilities::int_to_string(n_mpi_processes, 2) +
193  data_out->default_suffix());
194 
195  std::ofstream master_output ((current_name + ".pvtu").c_str());
196  data_out->write_pvtu_record (master_output, filenames);
197  }
198  }
199  data_out = 0;
200  deallog << "Reset output." << std::endl;
201  initialized = false;
202  output_file.close();
203  deallog.pop();
204 }
