43 int t_dae_residual(realtype tt, N_Vector yy, N_Vector yp,
44  N_Vector rr, void *user_data)
45 {
46  SundialsInterface<VEC> &solver = *static_cast<SundialsInterface<VEC> *>(user_data);
47 
48  shared_ptr<VEC> src_yy = solver.create_new_vector();
49  shared_ptr<VEC> src_yp = solver.create_new_vector();
50  shared_ptr<VEC> residual = solver.create_new_vector();
51 
52  copy(*src_yy, yy);
53  copy(*src_yp, yp);
54 
55  int err = solver.residual(tt, *src_yy, *src_yp, *residual);
56 
57  copy(rr, *residual);
58 
59  return err;
60 }

97  N_Vector b,
98  N_Vector weight,
99  N_Vector yy,
100  N_Vector yp,
101  N_Vector resp)
102 {
103  (void) weight;
104  SundialsInterface<VEC> &solver = *static_cast<SundialsInterface<VEC> *>(IDA_mem->ida_user_data);
105 
106  shared_ptr<VEC> src_yy = solver.create_new_vector();
107  shared_ptr<VEC> src_yp = solver.create_new_vector();
108  shared_ptr<VEC> residual = solver.create_new_vector();
109  shared_ptr<VEC> dst = solver.create_new_vector();
110  shared_ptr<VEC> src = solver.create_new_vector();
111 
112  copy(*src_yy, yy);
113  copy(*src_yp, yp);
114  copy(*residual, resp);
115  copy(*src, b);
116 
117  int err = solver.solve_jacobian_system(IDA_mem->ida_tn,
118  *src_yy,
119  *src_yp,
120  *residual,
121  IDA_mem->ida_cj,
122  *src,
123  *dst);
124  copy(b, *dst);
125  return err;

232 template <typename VEC>
233 unsigned int IDAInterface<VEC>::start_ode(VEC &solution,
234  VEC &solution_dot,
235  const unsigned int max_steps)
236 {
237 
238 
239  AssertThrow(solution.size() == solver.n_dofs(),
240  ExcDimensionMismatch(solution.size(), solver.n_dofs()));
241 
242  AssertThrow(is_initialized, ExcMessage("Not Initialized!"));
243 
244  double t = initial_time;
245  double h = initial_step_size;
246  unsigned int step_number = 0;
247 
248  int status;
249 
250  // The solution is stored in
251  // solution. Here we take only a
252  // view of it.
253 
254 #ifdef DEAL_II_WITH_MPI
255  IndexSet is = solution.locally_owned_elements();
256 
257  yy = N_VNew_Parallel(solver.get_comm(), is.n_elements(), solver.n_dofs());
258  yp = N_VNew_Parallel(solver.get_comm(), is.n_elements(), solver.n_dofs());
259  diff_id = N_VNew_Parallel(solver.get_comm(), is.n_elements(), solver.n_dofs());
260  abs_tolls = N_VNew_Parallel(solver.get_comm(), is.n_elements(), solver.n_dofs());
261 #else
262  yy = N_VNew_Serial(solver.n_dofs());
263  yp = N_VNew_Serial(solver.n_dofs());
264  diff_id = N_VNew_Serial(solver.n_dofs());
265  abs_tolls = N_VNew_Serial(solver.n_dofs());
266 #endif
267 
268  reset_ode(initial_time, solution, solution_dot, initial_step_size, max_steps, true);
269 
270  double next_time = 0;
271 
272  solver.output_step( 0, solution, solution_dot, 0, initial_step_size);
273 
274  while ((t<final_time) && (step_number < max_steps))
275  {
276 
277  next_time += outputs_period;
278  if (verbose)
279  {
280  pout << " "//"\r"
281  << std::setw(5) << t << " ----> "
282  << std::setw(5) << next_time
283  << std::endl;
284  }
285  status = IDASolve(ida_mem, next_time, &t, yy, yp, IDA_NORMAL);
286 
287  status = IDAGetLastStep(ida_mem, &h);
288  AssertThrow(status == 0, ExcMessage("Error in IDA Solver"));
289 
290  copy(solution, yy);
291  copy(solution_dot, yp);
292 
293  // Check the solution
294  bool reset = solver.solver_should_restart(t, step_number, h, solution, solution_dot);
295 
296 
297  while ( reset == true )
298  {
299  // double frac = 0;
300  int k = 0;
301  IDAGetLastOrder(ida_mem, &k);
302  // frac = std::pow((double)k,2.);
303  reset_ode(t, solution, solution_dot,
304  h/2.0, max_steps, false);
305  reset = solver.solver_should_restart(t, step_number, h, solution, solution_dot);
306  }
307 
308  step_number++;
309 
310  solver.output_step(t, solution, solution_dot, step_number, h);
311 
312 
313 
314  }
315 
316  pout << std::endl;
317  // Free the vectors which are no longer used.
318 #ifdef DEAL_II_WITH_MPI
319  N_VDestroy_Parallel(yy);
320  N_VDestroy_Parallel(yp);
321  N_VDestroy_Parallel(abs_tolls);
322  N_VDestroy_Parallel(diff_id);
323 #else
324  N_VDestroy_Serial(yy);
325  N_VDestroy_Serial(yp);
326  N_VDestroy_Serial(abs_tolls);
327  N_VDestroy_Serial(diff_id);
328 #endif
329 
330  return step_number;
331 }

333 template <typename VEC>
334 void IDAInterface<VEC>::reset_ode(double current_time,
335  VEC &solution,
336  VEC &solution_dot,
337  double current_time_step,
338  unsigned int max_steps,
339  bool first_step)
340 {
341  if (ida_mem)
342  IDAFree(&ida_mem);
343 
344  ida_mem = IDACreate();
345 
346 
347  // Free the vectors which are no longer used.
348  if (yy)
349  {
350 #ifdef DEAL_II_WITH_MPI
351  N_VDestroy_Parallel(yy);
352  N_VDestroy_Parallel(yp);
353  N_VDestroy_Parallel(abs_tolls);
354  N_VDestroy_Parallel(diff_id);
355 #else
356  N_VDestroy_Serial(yy);
357  N_VDestroy_Serial(yp);
358  N_VDestroy_Serial(abs_tolls);
359  N_VDestroy_Serial(diff_id);
360 #endif
361  }
362 
363  int status;
364  Assert(solution.size() == solver.n_dofs(),
365  ExcDimensionMismatch(solution.size(), solver.n_dofs()));
366 
367  Assert(solution_dot.size() == solver.n_dofs(),
368  ExcDimensionMismatch(solution_dot.size(), solver.n_dofs()));
369 
370 
371 #ifdef DEAL_II_WITH_MPI
372  IndexSet is = solution.locally_owned_elements();
373 
374  yy = N_VNew_Parallel(solver.get_comm(), is.n_elements(), solver.n_dofs());
375  yp = N_VNew_Parallel(solver.get_comm(), is.n_elements(), solver.n_dofs());
376  diff_id = N_VNew_Parallel(solver.get_comm(), is.n_elements(), solver.n_dofs());
377  abs_tolls = N_VNew_Parallel(solver.get_comm(), is.n_elements(), solver.n_dofs());
378 #else
379  yy = N_VNew_Serial(solver.n_dofs());
380  yp = N_VNew_Serial(solver.n_dofs());
381  diff_id = N_VNew_Serial(solver.n_dofs());
382  abs_tolls = N_VNew_Serial(solver.n_dofs());
383 #endif
384 
385  copy(yy, solution);
386  copy(yp, solution_dot);
387  copy(diff_id, solver.differential_components());
388 
389  status = IDAInit(ida_mem, t_dae_residual<VEC>, current_time, yy, yp);
390 
391  if (use_local_tolerances)
392  {
393  VEC &tolerances = solver.get_local_tolerances();
394  VEC abs_tolerances(tolerances);
395  abs_tolerances /= tolerances.linfty_norm();
396  abs_tolerances *= abs_tol;
397  copy(abs_tolls, abs_tolerances);
398  status += IDASVtolerances(ida_mem, rel_tol, abs_tolls);
399  }
400  else
401  {
402  status += IDASStolerances(ida_mem, rel_tol, abs_tol);
403  }
404 
405  status += IDASetInitStep(ida_mem, current_time_step);
406  status += IDASetUserData(ida_mem, (void *) &solver);
407 
408  status += IDASetId(ida_mem, diff_id);
409  status += IDASetSuppressAlg(ida_mem, ignore_algebraic_terms_for_errors);
410 
411  status += IDASetMaxNumSteps(ida_mem, max_steps);
412  status += IDASetStopTime(ida_mem, final_time);
413 
414  status += IDASetMaxNonlinIters(ida_mem, max_non_linear_iterations);
415 
416  // Initialize solver
417  IDAMem IDA_mem;
418  IDA_mem = (IDAMem) ida_mem;
419 
420  IDA_mem->ida_lsetup = t_dae_lsetup<VEC>;
421  IDA_mem->ida_lsolve = t_dae_solve<VEC>;
422  IDA_mem->ida_setupNonNull = true;
423 
424 
425  status += IDASetMaxOrd(ida_mem, max_order);
426 
427  AssertThrow(status == 0, ExcMessage("Error initializing IDA."));
428 
429  std::string type;
430  if (first_step)
431  type = ic_type;
432  else
433  type = reset_type;
434 
435  if (type == "use_y_dot")
436  {
437  // (re)initialization of the vectors
438  IDACalcIC(ida_mem, IDA_Y_INIT, current_time+current_time_step);
439  IDAGetConsistentIC(ida_mem, yy, yp);
440 
441  copy(solution, yy);
442  copy(solution_dot, yp);
443  }
444  else if (type == "use_y_diff")
445  {
446  IDACalcIC(ida_mem, IDA_YA_YDP_INIT, current_time+current_time_step);
447  IDAGetConsistentIC(ida_mem, yy, yp);
448 
449  copy(solution, yy);
450  copy(solution_dot, yp);
451  }
452 }
