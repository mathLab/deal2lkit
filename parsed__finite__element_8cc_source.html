71 template<int dim, int spacedim>
72 void ParsedFiniteElement<dim,spacedim>::parse_parameters_call_back()
73 {
74  component_blocks.resize(component_names.size());
75  block_names.resize(component_names.size());
76  unsigned int j=0;
77  for (unsigned int i=0; i<component_names.size(); ++i)
78  {
79  if ( (i>0) && (component_names[i-1] != component_names[i]) )
80  j++;
81  component_blocks[i] = j;
82  block_names[j] = component_names[i];
83  }
84  block_names.resize(j+1);
85  FiniteElement<dim,spacedim> *fe = (*this)();
86  const unsigned int nc = fe->n_components();
87  delete fe;
88  AssertThrow(component_names.size() == nc,
89  ExcInternalError("Generated FE has the wrong number of components."));
90 }
