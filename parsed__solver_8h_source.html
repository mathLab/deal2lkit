206 template<typename VECTOR>
207 void ParsedSolver<VECTOR>::parse_parameters_call_back()
208 {
209  if (solver_name == "cg")
210  {
211  initialize_solver(new SolverCG<VECTOR>(control));
212  }
213  else if (solver_name == "bicgstab")
214  {
215  initialize_solver(new SolverBicgstab<VECTOR>(control));
216  }
217  else if (solver_name == "gmres")
218  {
219  initialize_solver(new SolverGMRES<VECTOR>(control));
220  }
221  else if (solver_name == "fgmres")
222  {
223  initialize_solver(new SolverFGMRES<VECTOR>(control));
224  }
225  else if (solver_name == "minres")
226  {
227  initialize_solver(new SolverMinRes<VECTOR>(control));
228  }
229  else if (solver_name == "qmrs")
230  {
231  initialize_solver(new SolverQMRS<VECTOR>(control));
232  }
233  else if (solver_name == "richardson")
234  {
235  initialize_solver(new SolverRichardson<VECTOR>(control));
236  }
237  else
238  {
239  Assert(false, ExcInternalError("Solver should not be unknonw."));
240  }
241 }
