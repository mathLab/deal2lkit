171  {
172  const unsigned int n_dofs = get_current_fe_values().get_fe().dofs_per_cell;
173

194  template<typename Number>
195  const std::vector <std::vector<Number> > &
196  get_values(const std::string &prefix, const Number dummy)
197  {
198  const std::vector<Number> &independent_local_dofs =
199  get_current_independent_local_dofs(prefix, dummy);
200 
201  const FEValuesBase<dim,spacedim> &fev = get_current_fe_values();
202 
203  const unsigned int dofs_per_cell = fev.dofs_per_cell;
204  const unsigned int n_q_points = fev.n_quadrature_points;
205  const unsigned int n_components = fev.get_fe().n_components();
206 
207  std::string name = prefix+"_all_values_q"+Utilities::int_to_string(n_q_points)+"_"+type(dummy);
208 
209  if (!cache.have(name))
210  {
211  cache.add_copy(std::vector <std::vector<Number> >(n_q_points, std::vector<Number>(n_components)), name);
212  }
213 
214  std::vector <std::vector<Number> > &ret = cache.template get<std::vector <std::vector<Number> > >(name);
215  DOFUtilities::get_values(fev, independent_local_dofs, ret);
216  return ret;
217  }
